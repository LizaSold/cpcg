\newglossaryentry{MRI}{
	name={magnetic resonance imaging},
	description={non-invasive imaging technique including the use of high magnetic induction (~$ 1-7T $)}
}

\newglossaryentry{DMRI}{
	name={Diffusion Weighted \Gls{MRI}},
	description={imaging technique to determine diffusion of water in body tissues using \acrshort{mri}}
}

\newglossaryentry{FA}{
	name={fractional anisotropy},
	description={scalar metric for the directionality of diffusion; high values point to a constrained movement of particles like in an axon; low values point to a free movement of particles like in grey matter}
}

\newglossaryentry{DTI}{
	name={diffusion tensor imaging},
	description={imaging technique to determine main directions and strengths of water diffusion in tissue per voxel using \acrshort{dmri}}
}

\newglossaryentry{CSD}{
	name={Constrained Spherical Deconvolution},
	description={tba}
}

\newglossaryentry{AD}{
	name={Alzheimers Disease},
	description={neurodegenerative disease, cf. \gls{AMYHYP}}
}

\newglossaryentry{NIH}{
	name={National Institutes of Health},
	description={''The National Institutes of Health (NIH), a part of the U.S. Department of Health and Human Services, is the nation's medical research agency -- making important discoveries that improve health and save lives.''\newcite{nih_about}}
}

\newglossaryentry{HCP}{
	name={Human Connectome Project},
	description={''Launched in 2009 as a Blueprint Grand Challenge, the NIH Human Connectome Project (HCP) is an ambitious effort to map the neural pathways that underlie human brain function. The overarching purpose of the Project is to acquire and share data about the structural and functional connectivity of the human brain.''\newcite{nih_connectome}}
}

\newglossaryentry{CCF}{
	name={Connectome Coordination Facility},
	description={''The Connectome Coordination Facility (CCF) houses and distributes public research data for [\acrshort{hcp}].''\newcite{ccf}}
}

\newglossaryentry{LHCP}{
	name={Lifespan \acrshort{hcp}},
	description={''HCP Lifespan Projects are acquiring and sharing multimodal imaging data acquired across the lifespan, in four age groups (prenatal, 0-5, 6-21, and 36-100+).''\newcite{ccf}}
}

\newglossaryentry{AMYHYP}{
	name={Amyloid Hypothesis},
	description={hypothesis naming the accumulation of amyloid-$ \beta $ peptide A$ \beta $42 in structures of the hippocampus, forebrain and neocortex as cause of \Gls{AD}}
}

\newglossaryentry{FBA}{
	name={fixel-based analysis},
	description={method for analyzing DWI data using constraint spherical deconvolution; ''fixel'' refers to a specific fibre population within a voxel; see \cite{ad}}
}

\newglossaryentry{ADNI}{
	name={Alzheimers Disease Neuroimaging Initiative},
	description={''longitudinal multicenter study designed to develop clinical, imaging, genetic, and biochemical biomarkers for the early detection and tracking of \acrfull{ad}''\newcite{adni_about}}
}
